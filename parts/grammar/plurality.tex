\section{Plurality}
Plurality marking is optional.
If the speaker does not know the quantity of a noun, they may simply omit it.

\begin{table}[h!]
    \centering
    \caption{Plurality}
    \begin{tabularx}{8cm}{|X|X|}
        \hline
        \textbf{Plurality} & \textbf{Suffix} \\
        \hline
        None (zero) & -pa \\
        Singular & -sa \\
        Dual & -gula \\
        Plural & -elah \\
        \hline
    \end{tabularx}
\end{table}
